  ```
import math
def distance(x1,y1,x2,y2):
     dx=x2-x1
     dy=y2-y1
     d=math.sqrt(dx**2+dy**2)
     return d
x1=4
y1=2
x2=10
y2=6
d=distance(x1,y1,x2,y2)
print(round(d,2))


  ```
